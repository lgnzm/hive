  id            string,
  datetimes     string,
  monthname     string,
  yearnumber    string,
  monthnumber   string,
  daynumber     string,
  hournumber    string,
  minutenumber  string,
  ampm          string
)

INSERT INTO TABLE assessdb.rawdatetime001
SELECT id, datetimes, monthname, yeartimes, monthnumber; daynumber, hournumber, minutenumber, ampm
FROM retrivedb.rawdatetime;
WHERE id <> '"id"';

  monthname     string,
  yearnumber    string,
  monthnumber   string,
  daynumber     string,
  hournumber    string,
  minutenumber  string,

)

INSERT INTO TABLE assessdb.rawdatetime002
